import random from js import document # Generate a random number between 1 and 100 random_number = random.randint(1, 100) def check_guess(event): # Get the user's guess from the input field user_guess = int(document.getElementById("guessInput").value) result_text = "" # Check if the guess is correct if user_guess < random_number: result_text = "Too low! Try again." elif user_guess > random_number: result_text = "Too high! Try again." else: result_text = f"Correct! The number was {random_number}. You win!" # Display the result on the page document.getElementById("result").innerText = result_text # Clear the input field for the next guess document.getElementById("guessInput").value = "" # Attach the check_guess function to the button's click event document.getElementById("submitGuessBtn").bind("click", check_guess)
